39private:
40 static bool roughly_equals(const date_time& time, const date_time& time_with_window, int window_in_seconds, int frequency_in_seconds) {
41 auto delta = convert::to_int32((time_with_window - time).total_seconds_duration().count()) % frequency_in_seconds;
42
43 delta = delta > window_in_seconds ? frequency_in_seconds - delta : delta;
44
45 return math::abs(delta) < window_in_seconds;
46 }
