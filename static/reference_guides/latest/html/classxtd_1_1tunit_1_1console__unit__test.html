The console_unit_test class is console unit test interface.

Namespace

Library

Remarks
All messages are displayed on the console.
Examples
This example shows how to instantiate and launch tests with console unit test.
#include <xtd/xtd.tunit>
#include <sstream>
using namespace xtd;
using namespace xtd::tunit;
namespace unit_tests {
class test_class_(test) {
void test_method_(test_case1) {
}
};
}
auto main() -> int {
return console_unit_test().run();
// is same :
//return unit_test(new_uptr<ostream_event_listener>(console::out)).run();
}
// This code can produce the following output :
//
// Start 1 test from 1 test case
// Run tests:
// SUCCEED unit_tests::test.test_case1 (0 ms total)
//
// Test results:
// SUCCEED 1 test.
// End 1 test from 1 test case ran. (0 ms total)
