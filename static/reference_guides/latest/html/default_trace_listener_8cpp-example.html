default_trace_listener.cpp

Shows how to use xtd::diagnostics::default_trace_listener class.

1#include <xtd/diagnostics/trace>
2#include <xtd/diagnostics/default_trace_listener>
3#include <xtd/console>
4#include <xtd/startup>
5
6using namespace xtd;
7using namespace xtd::collections::generic;
8using namespace xtd::diagnostics;
9
10class binomial {
11public:
12 // args(0) is the number of possibilities for binomial coefficients.
13 // args(1) is the file specification for the trace log file.
14 static auto main(const list<ustring>& args) {
15 auto possibilities = .0l;
16 auto iter = .0l;
17
18 // Remove the original default trace listener.
19 trace::listeners().erase(trace::listeners().begin());
20
21 // Create and add a new default trace listener.
22 auto default_listener = new_ptr<default_trace_listener>();
23 trace::listeners().push_back(default_listener);
24
25 // Assign the log file specification from the command line, if entered.
26 if (args.size() >= 2)
27 default_listener->log_file_name(args[1]);
28
29 // Validate the number of possibilities argument.
30 if (args.size() >= 1) {
31 // Verify that the argument is a number within the correct range.
32 try {
33 const auto MAX_POSSIBILITIES = 99.0l;
34 possibilities = parse<decimal>(args[0]);
35 if (possibilities < 0 || possibilities > MAX_POSSIBILITIES)
36 throw new system_exception(ustring::format("The number of possibilities must be in the range 0..{}.", MAX_POSSIBILITIES));
37 } catch (const system_exception& ex) {
38 auto fail_message = ustring::format("\"{}\" is not a valid number of possibilities.", args[0]);
39 default_listener->fail(fail_message, ex.message());
40 if (!default_listener->assert_ui_enabled())
41 console::write_line(fail_message + "\n" + ex.message());
42 return;
43 }
44 } else {
45 // Report that the required argument is not present.
46 const auto ENTER_PARAM = "Enter the number of possibilities as a command line argument.";
47 default_listener->fail(ENTER_PARAM);
48 if (!default_listener->assert_ui_enabled())
49 console::write_line(ENTER_PARAM);
50 return;
51 }
52
53 for (iter = 0; iter <= possibilities; iter++) {
54 auto result = .0l;
55 auto binomial = ustring::empty_string;
56
57 // Compute the next binomial coefficient and handle all exceptions.
58 try {
59 result = calc_binomial(possibilities, iter);
60 } catch (const system_exception& ex) {
61 auto fail_message = ustring::format("An exception was raised when calculating Binomial( {}, {} ).", possibilities, iter);
62 default_listener->fail(fail_message, ex.message());
63 if (!default_listener->assert_ui_enabled())
64 console::write_line(fail_message + "\n" + ex.message());
65 return;
66 }
67
68 // Format the trace and console output.
69 binomial = ustring::format("Binomial( {0}, {1} ) = ", possibilities, iter);
70 default_listener->write(binomial);
71 default_listener->write_line(ustring::format("{}", result));
72 console::write_line("{0} {1}", binomial, result);
73 }
74 }
75
76 static decimal calc_binomial(xtd::decimal possibilities, decimal outcomes) {
77 // Calculate a binomial coefficient, and minimize the chance of overflow.
78 auto result = 1.0l;
79 auto iter = .0l;
80 for (iter = 1.0l; iter <= possibilities - outcomes; iter++) {
81 result *= outcomes + iter;
82 result /= iter;
83 }
84 return result;
85 }
86};
87
88startup_(binomial::main);
