13class test_ip_address {
14public:
15 static auto main(const list<ustring>& args) {
16 auto server = ustring::empty_string;
17
18 // Define a regular expression to parse user's input.
19 // This is a security check. It allows only
20 // alphanumeric input string between 2 to 40 character long.
21 auto rex = std::regex {R"(^[a-zA-Z]\w{1,39}$)"};
22
23 if (args.size() < 1) {
24 // If no server name is passed as an argument to this program, use the current
25 // server name as default.
26 server = dns::get_host_name();
27 console::write_line("Using current host: {}", server);
28 } else {
29 server = args[0];
30 if (!regex_match(server, rex)) {
31 console::write_line("Input string format not allowed.");
32 return;
33 }
34 }
35
36 // Get the list of the addresses associated with the requested server.
37 ip_addresses(server);
38
39 // Get additional address information.
40 ip_address_additional_info();
41 }
42
43private:
49 static void ip_addresses(const ustring& server) {
50 try {
51 // Get server related information.
52 auto heserver = dns::get_host_entry(server);
53
54 // Loop on the AddressList
55 for (auto cur_add : heserver.address_list()) {
56 // Display the type of address family supported by the server. If the
57 // server is IPv6-enabled this value is: InterNetworkV6. If the server
58 // is also IPv4-enabled there will be an additional value of InterNetwork.
59 console::write_line("address_family: {}", cur_add.address_family());
60
61 // Display the ScopeId property in case of IPV6 addresses.
62 if (cur_add.address_family() == address_family::inter_network_v6)
63 console::write_line("scope id: {}", cur_add.scope_id());
64
65
66 // Display the server IP address in the standard format. In
67 // IPv4 the format will be dotted-quad notation, in IPv6 it will be
68 // in in colon-hexadecimal notation.
69 console::write_line("address: {}", cur_add);
70
71 // Display the server IP address in byte format.
72 console::write("address_bytes: ");
73
74 auto bytes = cur_add.get_address_bytes();
75 for (size_t i = 0; i < bytes.size(); i++)
76 console::write(bytes[i]);
77 console::write_line("\r\n");
78 }
79 } catch (const system_exception& e) {
80 console::write_line("[do_resolve] exception: {}", e);
81 }
82 }
83
84 // This IPAddressAdditionalInfo displays additional server address information.
85 static void ip_address_additional_info() {
86 try {
87 // Display the flags that show if the server supports IPv4 or IPv6
88 // address schemas.
89 console::write_line("\r\nos_supports_ip_v4: {}", socket::os_supports_ip_v4());
90 console::write_line("os_supports_ip_v6: {}", socket::os_supports_ip_v6());
91
92 if (socket::os_supports_ip_v6()) {
93 // Display the server Any address. This IP address indicates that the server
94 // should listen for client activity on all network interfaces.
95 console::write_line("\r\nip_v6_any: {}", ip_address::ip_v6_any);
96
97 // Display the server loopback address.
98 console::write_line("ip_v6_loopback: {}", ip_address::ip_v6_loopback);
99
100 // Used during autoconfiguration first phase.
101 console::write_line("ip_v6_none: {}", ip_address::ip_v6_none);
102
103 console::write_line("is_loopback(ip_v6_loopback): {}", ip_address::is_loopback(ip_address::ip_v6_loopback));
104 }
105 console::write_line("is_loopback(loopback):{}", ip_address::is_loopback(ip_address::loopback));
106 } catch (const system_exception& e) {
107 console::write_line("[ip_addresses] exception: {}", e);
108 }
109 }
110};
