Shows how to use xtd::timers::timer class.

9namespace timer_example {

15 console::write_line("\nPress the Enter key to exit the application...\n");
16 console::write_line("The application started at {:t}.{:d3}", date_time::now(), date_time::now().millisecond());

18 a_timer.stop();

24 static void set_timer() {
25 // Hook up the elapsed event for the timer.
26 a_timer.elapsed += on_timed_event;
27 a_timer.auto_reset(true);
28 a_timer.enabled(true);
29 }
30
31 static void on_timed_event(object& source, const elapsed_event_args& e) {
32 console::write_line("The elapsed event was raised at {:t}.{:d3}",
33 e.signal_time(), e.signal_time().millisecond());
34 }
35
36 private:
37 // Create a timer with a two second interval.
