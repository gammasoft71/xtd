#include <xtd/drawing/color_converter>
#include <xtd/forms/application>
#include <xtd/forms/form>
#include <xtd/forms/lcd_label>
#include <xtd/forms/timer>
using namespace xtd;
using namespace xtd::drawing;
using namespace xtd::forms;
auto main() -> int {
auto index = 0_z;
form form_main;
form_main.text("Lcd label with dot matrix example");
form_main.client_size({403, 403});
lcd_label label;
label.parent(form_main);
label.dock(dock_style::fill);
label.fore_color(color::yellow);
label.back_color(color_converter::average(color::black, label.fore_color(), 0.05));
label.lcd_style(forms::lcd_style::dot_matrix_display);
label.text(string::format("{}", label.valid_characters()[index]));
timer timer;
timer.interval(300_ms);
timer.enabled(!timer.enabled());
timer.tick += [&] {
if (++index >= label.valid_characters().size()) index = 0;
label.text(string::format("{}", label.valid_characters()[index]));
};
application::run(form_main);
}
