Shows how to use xtd::threading::thread class.

2#include <xtd/console>
3#include <xtd/startup>

5using namespace xtd;
6using namespace xtd::threading;
7
8// Simple threading scenario: Start a static method running
9// on a second thread.
10class thread_example {
11public:
12 // The thread_proc method is called when the thread starts.
13 // It loops ten times, writing to the console and yielding
14 // the rest of its time slice each time, and then ends.
15 static void thread_proc() {
16 for (auto i = 0; i < 10; ++i) {
17 console::write_line("thread_proc: {0}", i);
18 // Yield the rest of the time slice.
19 thread::sleep(0);
20 }
21 }
22
23 static void main() {
24 console::write_line("Main thread: Start a second thread.");
25 // The constructor for the thread class requires a thread_start
26 // delegate that represents the method to be executed on the
27 // thread. xtd simplifies the creation of this delegate.
28 auto t = thread {thread_proc};
29
30 // Start thread_proc. Note that on a uniprocessor, the new
31 // thread does not get any processor time until the main thread
32 // is preempted or yields. Uncomment the thread::sleep that
33 // follows t.start() to see the difference.
34 t.start();
35 //thread::sleep(0);
36
37 for (auto i = 0; i < 4; ++i) {
38 console::write_line("Main thread: Do some work.");
39 thread::sleep(0);
40 }
41
42 console::write_line("Main thread: Call join(), to wait until thread_proc ends.");
43 t.join();
44 console::write_line("Main thread: thread_proc.join has returned. Press Enter to end program.");
45 console::read_line();
46 }
47};
48
49startup_(thread_example::main);
50
