console_title.cpp

Shows how to use xtd::console::title method.

1#include <xtd/console>
2
3using namespace xtd;
4
5auto main() -> int {
6 console::write_line("The current console title is: \"{0}\"", console::title());

8 console::read_key(true);
9 console::title("The title has changed!");
10 console::write_line("Note that the new console title is \"{0}\"\n"
11 " (Press any key to quit.)", console::title());

13}
