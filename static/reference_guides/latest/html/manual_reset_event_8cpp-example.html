Shows how to use xtd::threading::manual_reset_event class.

7using namespace xtd;
8using namespace xtd::collections::generic;
9using namespace xtd::threading;

11namespace manual_reset_event_example {
12 class program {
13 public:
14 static void main() {
15 console::write_line("\nStart 3 named threads that block on a ManualresetEvent:\n");
16
17 for(auto i = 0; i <= 2; ++i) {
18 threads.emplace_back(thread_proc);
19 threads.back().name(ustring::format("Thread_{}", i));
20 threads.back().start();
21 }
22
23 thread::sleep(500);
24 console::write_line("\nWhen all three threads have started, press Enter to call set()"
25 "\nto release all the threads.\n");
26 console::read_line();
27
28 mre.set();
29
30 thread::sleep(500);
31 console::write_line("\nWhen a ManualresetEvent is signaled, threads that call WaitOne()"
32 "\ndo not block. Press Enter to show this.\n");
33 console::read_line();
34
35 for(auto i = 3; i <= 4; ++i) {
36 threads.emplace_back(thread_proc);
37 threads.back().name(ustring::format("Thread_{}", i));
38 threads.back().start();
39 }
40
41 thread::sleep(500);
42 console::write_line("\nPress Enter to call reset(), so that threads once again block"
43 "\nwhen they call WaitOne().\n");
44 console::read_line();
45
46 mre.reset();
47
48 // Start a thread that waits on the ManualresetEvent.
49 threads.emplace_back(thread_proc);
50 threads.back().name("Thread_5");
51 threads.back().start();
52
53 thread::sleep(500);
54 console::write_line("\nPress Enter to call set() and conclude the demo.");
55 console::read_line();
56
57 mre.set();
58 }
59
60 private:
61 inline static list<thread> threads = list<thread>(4);
62
63 // mre is used to block and release threads manually. It is
64 // created in the unsignaled state.
65 inline static manual_reset_event mre {false};
66
67 static void thread_proc() {
68 ustring name = thread::current_thread().name();
69
70 console::write_line(name + " starts and calls mre.WaitOne()");
71
72 mre.wait_one();
73
74 console::write_line(name + " ends.");
75 }
76 };
77}
