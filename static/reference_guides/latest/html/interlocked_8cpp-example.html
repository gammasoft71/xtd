interlocked.cpp

Shows how to use xtd::threading::interlocked class.

1#include <xtd/threading/interlocked>
2#include <xtd/threading/thread>
3#include <xtd/console>
4#include <xtd/random>
5#include <xtd/startup>
6
7using namespace xtd;
8using namespace xtd::threading;
9
10namespace interlocked_example {
11 struct my_thread {
12 threading::thread thread;
13 ustring name;
14 };
15
16 class my_interlocked_exchange_class {
17 private:
18 //0 for False, 1 for True.
19 inline static int using_resource = 0;
20
21 static constexpr int num_thread_iterations = 5;
22
23 public:
24 // The main entry point for the application.
25 static auto main() {
26 auto my_threads = std::array<my_thread, 10> {};
27 auto rnd = xtd::random {};
28
29 for (auto index = 0ul; index < my_threads.size(); ++index) {
30 my_threads[index].name = ustring::format("Thread{}", index + 1);
31
32 //Wait a random amount of time before starting next thread.
33 thread::sleep(rnd.next(0, 1000));
34 my_threads[index].thread = thread {my_thread_proc};
35 my_threads[index].thread.start(my_threads[index].name);
36 }
37
38 for (auto index = 0ul; index < my_threads.size(); ++index)
39 my_threads[index].thread.join();
40 }
41
42 static void my_thread_proc(std::any name) {
43 for (auto index = 0; index < num_thread_iterations; ++index) {
44 use_resource(as<ustring>(name));
45
46 //Wait 1 second before next attempt.
47 thread::sleep(1000);
48 }
49 }
50
51 //A simple method that denies reentrancy.
52 static auto use_resource(const ustring& name)->bool {
53 //0 indicates that the method is not in use.
54 if (0 == interlocked::exchange(using_resource, 1)) {
55 console::write_line("{} acquired the lock", name);
56
57 //Code to access a resource that is not thread safe would go here.
58
59 //Simulate some work
60 thread::sleep(500);
61
62 console::write_line("{} exiting lock", name);
63
64 //Release the lock
65 interlocked::exchange(using_resource, 0);
66 return true;
67 } else {
68 console::write_line(" {} was denied the lock", name);
69 return false;
70 }
71 }
72 };
73}
74
75startup_(interlocked_example::my_interlocked_exchange_class::main);
