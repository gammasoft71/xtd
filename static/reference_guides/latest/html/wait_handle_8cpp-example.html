wait_handle.cpp

Shows how to use xtd::threading::wait_handle class.

1#include <xtd/threading/auto_reset_event>
2#include <xtd/threading/thread_pool>
3#include <xtd/console>
4#include <xtd/date_time>
5#include <xtd/random>
6#include <xtd/startup>
7
8using namespace xtd;
9using namespace xtd::threading;
10
11namespace wait_handle_example {
12 class program {
13 public:
14 static void main() {
15 // Queue up two tasks on two different threads;
16 // wait until all tasks are completed.
17 auto dt = date_time::now();
18 console::write_line("Main thread is waiting for BOTH tasks to complete.");
19 thread_pool::queue_user_work_item(do_task, event1);
20 thread_pool::queue_user_work_item(do_task, event2);
21 wait_handle::wait_all({event1, event2});
22 // The time shown below should match the longest task.
23 console::write_line("Both tasks are completed (time waited={0})",
24 date_time::now() - dt);
25
26 // Queue up two tasks on two different threads;
27 // wait until any task is completed.
28 dt = date_time::now();
29 console::write_line();
30 console::write_line("The main thread is waiting for either task to complete.");
31 thread_pool::queue_user_work_item(do_task, event1);
32 thread_pool::queue_user_work_item(do_task, event2);
33 auto index = wait_handle::wait_any({event1, event2});
34 // The time shown below should match the shortest task.
35 console::write_line("Task {0} finished first (time waited={1}).",
36 index + 1, date_time::now() - dt);
37 }
38
39 static void do_task(std::any state) {
40 auto are = as<auto_reset_event>(state);
41 auto time = 1000 * r.next(2, 10);
42 console::write_line("Performing a task for {0} milliseconds.", time);
43 thread::sleep(time);
44 are.set();
45 }
46
47 private:
48 // Define two auto_reset_event.
49 inline static auto_reset_event event1 {false};
50 inline static auto_reset_event event2 {false};
51
52 // Define a random number generator for testing.
53 inline static xtd::random r;
54 };
55}
56
57startup_(wait_handle_example::program::main);
