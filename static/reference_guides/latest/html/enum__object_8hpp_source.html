251 static enum_type parse_flags(const xtd::string& value, bool ignore_case) {
252 std::vector<xtd::string> values = value.split(',');
253 for (xtd::string& str : values)
254 str = str.trim(' ');
255
256 if (values.size() == 1) {
257 for (auto item : enum_object<enum_type>().entries()) {
258 if (xtd::string::compare(value, item.second, ignore_case) == 0)
259 return to_enum(item.first);
260 }
261 return to_enum(xtd::parse<int64>(value));
262 }
263
264 int64 result = 0;
265 for (xtd::string str : values) {
266 bool found = false;
267 for (auto item : enum_object<enum_type>().entries()) {
268 if (xtd::string::compare(str, item.second, ignore_case) == 0) {
269 found = true;
270 result |= to_int(item.first);
271 break;

274 if (found == false)

281 xtd::string to_string_flags() const noexcept {
282 auto iterator = std::find_if(entries().begin(), entries().end(), [&](auto value)->bool {return value.first == this->value;});
283 if (to_int(value) == 0 && iterator == entries().end()) return "0";
284
285 iterator = std::find_if(entries().begin(), entries().end(), [&](auto value)->bool {return value.first == to_enum(0);});
286 if (to_int(value) == 0) return iterator == entries().end() ? "0" : iterator->second;
287
288 xtd::string str;
289 int64 rest = to_int(value);
290 enum_collection<enum_type> reversed_entries = entries();
291 std::reverse(reversed_entries.begin(), reversed_entries.end());
292
293 for (auto item : reversed_entries) {
294 if (to_int(item.first) != 0 && (rest & to_int(item.first)) == to_int(item.first)) {
295 rest -= to_int(item.first);
296 if (!xtd::string::is_empty(str)) str = ", " + str;
297 str = item.second + str;
298 }
299 }
300
301 if (str.empty() || rest > 0) return xtd::string::format("{}", to_int(value));
302
303 return str;
