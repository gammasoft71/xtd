socket_tcp_ip_v4_without_thread.cpp

Shows how to use xtd::net::sockets::socket class.

8using namespace xtd;

11using namespace xtd::net::sockets;

14auto main() -> int {
15 auto terminate_app = false;
16
17 auto on_server_accept = [&](async_result ar) {
18 auto socket = as<xtd::net::sockets::socket>(ar->async_state()).end_accept(ar);
19
20 while (!terminate_app) {
21 auto buffer = list<unsigned char>(256);
22 auto number_of_byte_received = socket.receive(buffer);
23 if (number_of_byte_received) console::write_line(ustring {buffer.begin(), buffer.begin() + number_of_byte_received});
24 }
25 };
26
27 auto on_client_connect = [&](async_result ar) {
28 auto socket = as<xtd::net::sockets::socket>(ar->async_state());
29 socket.end_connect(ar);
30
31 auto counter = 0;
32 while (!terminate_app) {
33 auto str = ustring::format("socket={}, counter={}", socket.handle(), ++counter);
34 socket.send(list<unsigned char> {str.begin(), str.end()});
35 thread::sleep(50_ms);
36 }
37 };
38
39 const size_t client_count = 10;
40
41 auto server_socket = socket {address_family::inter_network, socket_type::stream, protocol_type::tcp};
42 server_socket.bind(ip_end_point {ip_address::any, 9400});
43 server_socket.listen();
44 for (auto index = 0_z; index < client_count; ++index)
45 server_socket.begin_accept(async_callback(on_server_accept), server_socket);
46
47 for (auto index = 0_z; index < client_count; ++index) {
48 auto socket = xtd::net::sockets::socket(address_family::inter_network, socket_type::stream, protocol_type::tcp);
49 socket.begin_connect(ip_address::loopback, 9400, async_callback(on_client_connect), socket);
50 }
51
52 console::read_key();
53 terminate_app = true;
54}
