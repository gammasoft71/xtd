28class location_tracker : public iobservable<location> {
29public:
30 location_tracker() = default;
31
32 void subscribe(iobserver<location>& observer) noexcept override {
33 if (find(observers_.begin(), observers_.end(), &observer) == observers_.end())
34 observers_.push_back(&observer);
35 }
36
37 void unsubscribe(iobserver<location>& observer) noexcept override {
38 auto iterator = find(observers_.begin(), observers_.end(), &observer);
39 if (iterator != observers_.end())
40 observers_.erase(iterator);
41 }
42
43 void track_location(std::optional<location> loc) {
44 for (auto observer : observers_) {
45 if (!loc.has_value())
46 observer->on_error(location_unknown_exception());
47 else
48 observer->on_next(loc.value());
49 }
50 }
51
52 void end_transmission() {
53 for (auto observer : observers_)
54 observer->on_completed();
55 observers_.clear();
56 }
57
58private:
59 list<iobserver<location>*> observers_;
60};

62class location_reporter : iobserver<location> {
63public:
64 location_reporter(ustring name) : name_(name) {}
65 virtual ~location_reporter() {unsubscribe();}
66
67 ustring name() const noexcept {return name_;}
68
69 virtual void subscribe(iobservable<location>& provider) {
70 provider_ = &provider;
71 provider_->subscribe(*this);
72 }
73
74 void on_completed() noexcept override {
75 console::write_line("The location Tracker has completed transmitting data to {}.", name());
76 unsubscribe();
77 }
78
79 void on_error(const std::exception& e) noexcept override {
80 console::write_line("{}: The location cannot be determined.", name());
81 }
82
83 void on_next(const location& value) noexcept override {
84 console::write_line("{}: The current location is {}, {}", name(), value.latitude(), value.longitude());
85 }
86
87 virtual void unsubscribe() {
88 if (provider_ != nullptr) provider_->unsubscribe(*this);
89 provider_ = nullptr;
90 }
91
92private:
93 ustring name_;
94 iobservable<location>* provider_ = nullptr;
95};
