console_cursor_visible.cpp

Shows how to use xtd::console::cursor_visible property.

1#include <xtd/console>
2
3using namespace xtd;
4
5auto main() -> int {
6 auto m1 = "\nThe cursor is {0}.\nType any text then press Enter. "
7 "Type '+' in the first column to show \n"
8 "the cursor, '-' to hide the cursor, "
9 "or lowercase 'x' to quit:"_s;
10 auto s = ustring::empty_string;
11
12 auto save_cursor_visibile = console::cursor_visible();
13 auto save_cursor_size = console::cursor_size();
14 console::cursor_visible(true); // Initialize the cursor to visible.
15 console::cursor_size(100); // Emphasize the cursor.
16
17 while (true) {
18 console::write_line(m1, ((console::cursor_visible() == true) ? "VISIBLE" : "HIDDEN"));
19 s = console::read_line();
20 if (ustring::is_empty(s) == false) {
21 if (s[0] == '+')
22 console::cursor_visible(true);
23 else if (s[0] == '-')
24 console::cursor_visible(false);
25 else if (s[0] == 'x')
26 break;
27 }
28 }
29 console::cursor_visible(save_cursor_visibile);
30 console::cursor_size(save_cursor_size);
31}
