7class program {

35 // date_time_kind::local and display the result.
36
37 my_dt = date_time::specify_kind(save_now, date_time_kind::local);

47 // display the value and kind() property of a date_time structure, the
48 // date_time structure converted to local time, and the date_time
49 // structure converted to universal time.
50
51 static void display(const ustring& title, const date_time& input_dt) {
52 auto disp_dt = input_dt;
53 auto dt_string = ustring::empty_string;
54
55 // display the original date_time.
56
57 dt_string = disp_dt.to_string("u");
58 console::write_line("{0} {1}, kind = {2}", title, dt_string, disp_dt.kind());
59
60 // Convert input_dt to local time and display the result.
61 // If input_dt.kind() is date_time_kind.Utc, the conversion is performed.
62 // If input_dt.kind() is date_time_kind::local, the conversion is not performed.
63 // If input_dt.kind() is date_time_kind::unspecified, the conversion is
64 // performed as if input_dt was universal time.
65
66 disp_dt = input_dt.to_local_time();
67 dt_string = disp_dt.to_string("u");
68 console::write_line(" to_local_time: {0}, kind = {1}", dt_string, disp_dt.kind());
69
70 // Convert input_dt to universal time and display the result.
71 // If input_dt.kind() is date_time_kind.Utc, the conversion is not performed.
72 // If input_dt.kind() is date_time_kind::local, the conversion is performed.
73 // If input_dt.kind() is date_time_kind::unspecified, the conversion is
74 // performed as if input_dt was local time.
75
76 disp_dt = input_dt.to_universal_time();
77 dt_string = disp_dt.to_string("u");
78 console::write_line(" to_universal_time: {0}, kind = {1}", dt_string, disp_dt.kind());
79 console::write_line();
80 }
81
82 // display the value and kind property for date_time::now() and date_time::utc_now().
83
84 static void display_now(const ustring& title, const date_time& input_dt) {
85 auto dt_string = input_dt.to_string("u");
86 console::write_line("{0} {1}, kind = {2}", title, dt_string, input_dt.kind());
87 }
