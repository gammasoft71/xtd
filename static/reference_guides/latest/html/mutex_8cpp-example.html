Shows how to use xtd::threading::mutex class.

1#include <xtd/collections/generic/list>
2#include <xtd/threading/mutex>
3#include <xtd/threading/thread>
4#include <xtd/console>
5#include <xtd/startup>
6
7using namespace xtd;
8using namespace xtd::collections::generic;
9using namespace xtd::threading;
10
11namespace mutex_example {
12 class program {
13 public:
14 static void main() {
15 // Create the threads that will use the protected resource.
16 for (auto i = 0; i < num_threads; ++i) {
17 threads.emplace_back(thread_proc);
18 threads.back().name(ustring::format("thread_{0}", i + 1));
19 threads.back().start();
20 }
21
22 thread::join_all(threads);
23 }
24
25 static void thread_proc() {
26 for(auto i = 0; i < num_iterations; ++i)
27 Use_resource();
28 }
29
30 private:
31 // This method represents a resource that must be synchronized
32 // so that only one thread at a time can enter.
33 static void Use_resource() {
34 // Wait until it is safe to enter.
35 console::write_line("{0} is requesting the mutex",
36 thread::current_thread().name());
37 mut.wait_one();
38
39 console::write_line("{0} has entered the protected area",
40 thread::current_thread().name());
41
42 // Place code to access non-reentrant resources here.
43
44 // Simulate some work.
45 thread::sleep(500);
46
47 console::write_line("{0} is leaving the protected area",
48 thread::current_thread().name());
49
50 // Release the mutex.
51 mut.release_mutex();
52 console::write_line("{0} has released the mutex",
53 thread::current_thread().name());
54 }
55
56 inline static list<thread> threads;
57 // Create a new mutex. The creating thread does not own the mutex.
58 inline static mutex mut;
59 inline static constexpr int num_iterations = 1;
60 inline static constexpr int num_threads = 3;
61 };
62}
63
64startup_(mutex_example::program::main);
