Shows how to use xtd::threading::monitor class.

12namespace monitor_lock_example {
13 class sync_resource : public object {
14 public:
15 // Use a monitor to enforce synchronization.
16 void access() {
17 lock_(*this) {
18 console::write_line("Starting synchronized resource access on thread #{0}",
19 thread::current_thread().managed_thread_id());
20 if (thread::current_thread().managed_thread_id() % 2 == 0)
21 thread::sleep(2000);
22
23 thread::sleep(200);
24 console::write_line("Stopping synchronized resource access on thread #{0}",
25 thread::current_thread().managed_thread_id());
26 }
27 }
28 };
29
30 class un_sync_resource : public object {
31 public:
32 // Do not enforce synchronization.
33 void access() {
34 console::write_line("Starting unsynchronized resource access on Thread #{0}",
35 thread::current_thread().managed_thread_id());
36 if (thread::current_thread().managed_thread_id() % 2 == 0)
37 thread::sleep(2000);
38
39 thread::sleep(200);
40 console::write_line("Stopping unsynchronized resource access on thread #{0}",
41 thread::current_thread().managed_thread_id());
42 }
43 };

45 class app {
46 private:
47 inline static int num_ops = 0;
48 inline static auto_reset_event ops_are_done {false};
49 inline static sync_resource sync_res;
50 inline static un_sync_resource un_sync_res;
51
52 public:
53 static void main() {
54 // Set the number of synchronized calls.
55 num_ops = 5;
56 for (int ctr = 0; ctr <= 4; ++ctr)
57 thread_pool::queue_user_work_item(sync_update_resource);
58
59 // Wait until this WaitHandle is signaled.
60 ops_are_done.wait_one();
61 console::write_line("\t\nAll synchronized operations have completed.\n");
62
63 // Reset the count for unsynchronized calls.
64 num_ops = 5;
65 for (int ctr = 0; ctr <= 4; ctr++)
66 thread_pool::queue_user_work_item(un_sync_update_resource);
67
68 // Wait until this WaitHandle is signaled.
69 ops_are_done.wait_one();
70 console::write_line("\t\nAll unsynchronized thread operations have completed.\n");
71 }
72
73 static void sync_update_resource(std::any state) {
74 // Call the internal synchronized method.
75 sync_res.access();
76
77 // Ensure that only one thread can decrement the counter at a time.
78 if (interlocked::decrement(num_ops) == 0)
79 // Announce to Main that in fact all thread calls are done.
80 ops_are_done.set();
81 }
82
83 static void un_sync_update_resource(std::any state) {
84 // Call the unsynchronized method.
85 un_sync_res.access();
86
87 // Ensure that only one thread can decrement the counter at a time.
88 if (interlocked::decrement(num_ops) == 0)
89 // Announce to Main that in fact all thread calls are done.
90 ops_are_done.set();
91 }
92 };
93}
94
95startup_(monitor_lock_example::app::main);
