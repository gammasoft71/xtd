udp_client_ip_v4.cpp

Shows how to use xtd::net::sockets::udp_client class.

14auto main() -> int {
15 auto terminate_app = false;
16
17 auto server = thread {[&] {
18 auto udp = udp_client {ip_end_point {ip_address::any, 9400}};
19 while (!terminate_app) {
20 ip_end_point incoming_end_point;
21 auto buffer = udp.receive(incoming_end_point);
22 if (buffer.size() && buffer[0] != 0xFF)
23 console::write_line(ustring(buffer.begin(), buffer.end()));
24 }
25 }};
26
27 auto client = thread {[&] {
28 auto udp = udp_client {};
29 auto counter = 0;
30 while (!terminate_app) {
31 auto str = ustring::format("counter={}", ++counter);
32 udp.send(list<unsigned char>(str.begin(), str.end()), str.size(), ip_end_point(ip_address::loopback, 9400));
33 thread::sleep(50_ms);
34 }
35 udp.send(list<unsigned char> {0xFF}, 1, ip_end_point(ip_address::loopback, 9400));
36 }};
37
38 server.start();
39 client.start();
40
41 console::read_key();
42 terminate_app = true;
43 server.join();
44 client.join();
45}
