8void gen_time_span_from_hours(double hours) {
9 // Crée un objet time_span et une chaîne ustring à partir
10 // d'un nombre d'heures.
11 time_span interval = time_span::from_hours(hours);
12 ustring timeInterval = interval.to_string();
13
14 // Remplit la fin de la chaîne ustring avec des espaces si elle
15 // ne contient pas de millisecondes.
16 size_t pIndex = timeInterval.index_of(':');
17 pIndex = timeInterval.index_of( '.', pIndex );
18 if (pIndex < 0) timeInterval += " ";
19
20 console::write_line( "{0,21}{1,26}", hours, timeInterval );
21}
