tcp_client_ip_v6.cpp

Shows how to use xtd::net::sockets::tcp_client class.

1#include <xtd/io/stream_reader>
2#include <xtd/net/sockets/tcp_client>
3#include <xtd/net/sockets/tcp_listener>
4#include <xtd/threading/thread>
5#include <xtd/console>
6
7using namespace xtd;
8using namespace xtd::io;
9using namespace xtd::net;
10using namespace xtd::net::sockets;
11using namespace xtd::threading;
12
13auto main() -> int {
14 auto terminate_app = false;
15
16 auto server = thread {[&] {
17 auto listener = tcp_listener {ip_end_point(ip_address::ip_v6_any, 9400)};
18 listener.start();
19 auto stream = listener.accept_tcp_client().get_stream();
20 auto reader = stream_reader {stream};
21
22 while (!terminate_app)
23 if (stream.data_available()) console::write_line(reader.read_line());
24 }};
25
26 auto client = thread {[&] {
27 auto client = tcp_client {address_family::inter_network_v6};
28 client.connect(ip_address::ip_v6_loopback, 9400);
29 auto stream = client.get_stream();
30 auto writer = stream_writer {stream};
31
32 auto counter = 0;
33 while (!terminate_app) {
34 writer.write_line(ustring::format("counter={}", ++counter));
35 thread::sleep(50_ms);
36 }
37 }};
38
39 server.start();
40 client.start();
41
42 console::read_key();
43 terminate_app = true;
44 server.join();
45 client.join();
46}
47
