7using namespace xtd;
8using namespace xtd::io;
9
10class program {
11public:
12 static auto main() {
13 auto fi = file_info {path::combine(path::get_temp_path(), "MyTest.txt")};
14
15 // This text is added only once to the file.
16 if (!fi.exists()) {
17 //Create a file to write to.
18 block_scope_(auto sw = fi.create_text()) {
19 sw.write_line("Hello");
20 sw.write_line("And");
21 sw.write_line("Welcome");
22 }
23 }
24
25 // This text will always be added, making the file longer over time
26 // if it is not deleted.
27 block_scope_(auto sw = fi.append_text()) {
28 sw.write_line("This");
29 sw.write_line("is Extra");
30 sw.write_line("Text");
31 }
32
33 //Open the file to read from.
34 block_scope_(auto sr = fi.open_text()) {
35 while (!sr.end_of_stream())
36 console::write_line(sr.read_line());
37 }
38 }
39};
40
41startup_(program::main);
