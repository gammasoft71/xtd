745 template<typename ...args_t>
746 static ustring format(const ustring& fmt, args_t&& ... args) {
747 ustring result;
748 size_t index = 0;
749 std::vector<__format_information<char>> formats;
750 auto begin_format_iterator = fmt.end();
751 auto end_format_iterator = fmt.end();
752 for (auto iterator = fmt.begin(); iterator != fmt.end(); ++iterator) {
753 if (*iterator == '{') {
754 if (++iterator == fmt.end())
755 __throw_ustring_format_exception_open_bracket();
756 if (*iterator == '{')
757 result += *iterator;
758 else {
759 begin_format_iterator = iterator;
760 while (iterator != fmt.end() && *iterator != '}') ++iterator;
761 if (iterator == fmt.end())
762 __throw_ustring_format_exception_open_bracket();
763 end_format_iterator = iterator;
764 __format_information<char> fi;
765 fi.location = result.size();
766 std::string format_str {begin_format_iterator, end_format_iterator};
767 if (format_str.size() == 0)
768 fi.index = index++;
769 else {
770 size_t index_alignment_separator = ustring(format_str).index_of(',');
771 size_t index_format_separator = ustring(format_str).index_of(u':');
772
773 if (index_alignment_separator != std::string::npos && index_format_separator != std::string::npos && index_alignment_separator > index_format_separator)
774 index_alignment_separator = std::string::npos;
775
776 if (index_alignment_separator != std::string::npos)
777 fi.alignment = format_str.substr(index_alignment_separator + 1, index_format_separator != std::string::npos ? index_format_separator - index_alignment_separator - 1 : std::string::npos);
778
779 if (index_format_separator != std::string::npos)
780 fi.format = format_str.substr(index_format_separator + 1);
781
782 if (index_alignment_separator == 0 || index_format_separator == 0)
783 fi.index = index++;
784 else {
785 std::string index_str;
786 if (index_alignment_separator != std::string::npos)
787 index_str = format_str.substr(0, index_alignment_separator);
788 else if (index_format_separator != std::string::npos)
789 index_str = format_str.substr(0, index_format_separator);
790 else
791 index_str = std::move(format_str);
792 try {
793 for (auto c : index_str)
794 if (!std::isdigit(c)) __throw_ustring_format_exception_start_colon();
795 fi.index = std::stoi(index_str);
796 } catch (...) {
797 __throw_ustring_format_exception_start_colon();
798 }
799 }
800 }
801 formats.push_back(fi);
802 }
803 } else if (*iterator == '}') {
804 if (++iterator == fmt.end()) {
805 __throw_ustring_format_exception_close_bracket();
806 break;
807 }
808 if (*iterator != '}') {
809 __throw_ustring_format_exception_close_bracket();
810 break;
811 }
812 result += *iterator;
813 } else
814 result += *iterator;
815 }
816
817 __ustring_extract_format_arg(result, formats, std::forward<args_t>(args)...);
818 return result.c_str();
819 }
