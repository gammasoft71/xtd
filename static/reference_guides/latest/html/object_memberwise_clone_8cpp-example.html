object_memberwise_clone.cpp

Shows hows how to use xtd::object::memberwise_clone method.

1#include <xtd/console>
2
3using namespace xtd;
4
5class id_info : public object {
6public:
7 int id_number = 0;
8
9 explicit id_info(int id_number) : id_number {id_number} {}
10};
11
12class person : public object {
13public:
14 int age = 0;
15 ptr<ustring> name = new_ptr<ustring>();
16 ptr<::id_info> id_info = new_ptr<::id_info>(0);
17
18 uptr<person> shallow_copy() {
19 return memberwise_clone<person>();
20 }
21
22 uptr<person> deep_copy() {
23 auto other = memberwise_clone<person>();
24 other->id_info = new_ptr<::id_info>(this->id_info->id_number);
25 other->name = new_ptr<ustring>(*name);
26 return other;
27 }
28};
29
30void display_values(const person p) {
31 console::write_line(" name: {0:s}, age: {1:d}", *p.name, p.age);
32 console::write_line(" value: {0:d}", p.id_info->id_number);
33}

36 // Create an instance of person and assign values to its fields.
