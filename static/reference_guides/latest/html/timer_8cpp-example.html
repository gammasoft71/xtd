timer.cpp

Shows how to use xtd::threading::timer class.

1#include <xtd/threading/auto_reset_event>
2#include <xtd/threading/timer>
3#include <xtd/console>
4#include <xtd/date_time>
5#include <xtd/startup>
6
7using namespace xtd;
8using namespace xtd::threading;
9
10namespace timer_example {
11 class status_checker {
12 private:
13 int invoke_count = 0;
14 int max_count = 0;
15
16 public:
17 status_checker(int count) {
18 invoke_count = 0;
19 max_count = count;
20 }
21
22 // This method is called by the timer delegate.
23 void check_status(std::any state_info) {
24 auto auto_event = as<auto_reset_event>(state_info);
25 auto now = date_time::now();
26 console::write_line("{:t}.{:D3} Checking status {,2}.",
27 now, now.millisecond(),
28 ++invoke_count);
29
30 if (invoke_count == max_count) {
31 // Reset the counter and signal the waiting thread.
32 invoke_count = 0;
33 auto_event.set();
34 }
35 }
36 };
37
38 class program {
39 public:
40 static void main() {
41 // Create an auto_reset_event to signal the timeout threshold in the
42 // timer callback has been reached.
43 auto auto_event = auto_reset_event {false};
44
45 auto status_checker = timer_example::status_checker {10};
46
47 // Create a timer that invokes check_status after one second,
48 // and every 1/4 second thereafter.
49 auto now = date_time::now();
50 console::write_line("{:t}.{:D3} Creating timer.\n",
51 now, now.millisecond());
52 auto state_timer = timer {{status_checker, &timer_example::status_checker::check_status},
53 auto_event, 1000, 250};
54
55 // When auto_event signals, change the period to every half second.
56 auto_event.wait_one();
57 state_timer.change(0, 500);
58 console::write_line("\nChanging period to .5 seconds.\n");
59
60 // When auto_event signals the second time, dispose of the timer.
61 auto_event.wait_one();
62 state_timer.close();
63 console::write_line("\nDestroying timer.");
64 }
65 };
66}
67
68startup_(timer_example::program::main);
