Shows how to use xtd::threading::barrier class.

1#include <xtd/threading/barrier>
2#include <xtd/threading/interlocked>
3#include <xtd/threading/thread_pool>
4#include <xtd/console>
5#include <xtd/startup>
6
7using namespace xtd;
8using namespace xtd::threading;
9
10namespace barrier_example {
11 class program {
12 public:
13 static void main() {
14 auto count = 0;
15
16 // Create a barrier with three participants
17 // Provide a post-phase action that will print out certain information
18 // And the third time through, it will throw an exception
19 auto barrier = threading::barrier {3, action<threading::barrier&> {[&](threading::barrier& b) {
20 console::write_line("Post-Phase action: count={0}, phase={1}", count, b.current_phase_number());
21 if (b.current_phase_number() == 2) throw system_exception("D'oh!", csf_);
22 }}};
23
24 // Nope -- changed my mind. Let's make it five participants.
25 barrier.add_participants(2);
26
27 // Nope -- let's settle on four participants.
28 barrier.remove_participant();
29
30 auto action = [&] {
31 interlocked::increment(count);
32 barrier.signal_and_wait(); // during the post-phase action, count should be 4 and phase should be 0
33 interlocked::increment(count);
34 barrier.signal_and_wait(); // during the post-phase action, count should be 8 and phase should be 1
35
36 // The third time, SignalAndWait() will throw an exception and all participants will see it
37 interlocked::increment(count);
38 try {
39 barrier.signal_and_wait();
40 } catch (const barrier_post_phase_exception& bppe) {
41 console::write_line("Caught barrier_post_phase_exception: {0}", bppe.message());
42 }
43
44 // The fourth time should be hunky-dory
45 interlocked::increment(count);
46 barrier.signal_and_wait(); // during the post-phase action, count should be 16 and phase should be 3
47 };
48
49 // Now launch 4 parallel actions to serve as 4 participants
50 for (auto index = 0; index < 4; ++index)
51 thread_pool::queue_user_work_item(action);
52
53 // It's good form to join all thread and close() a barrier when you're done with it.
54 thread::join_all();
55 barrier.close();
56 }
57 };
58}
59
60startup_(barrier_example::program::main);
61

63//
64// Post-Phase action: count=4, phase=0
65// Post-Phase action: count=8, phase=1
66// Post-Phase action: count=12, phase=2
67// Post-Phase action: count=16, phase=3
