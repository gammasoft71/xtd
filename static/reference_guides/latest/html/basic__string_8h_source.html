2458inline std::basic_string<target_t> __xtd_convert_to_string(std::basic_string<source_t>&& str) noexcept {
2459 auto out = std::basic_string<target_t> {};
2460 auto codepoint = 0u;
2461 for (const auto& character : str) {
2462 if (character >= 0xd800 && character <= 0xdbff)
2463 codepoint = ((character - 0xd800) << 10) + 0x10000;
2464 else {
2465 if (character >= 0xdc00 && character <= 0xdfff) codepoint |= character - 0xdc00;
2466 else codepoint = character;
2467
2468 if (codepoint <= 0x7f)
2469 out.append(1, static_cast<target_t>(codepoint));
2470 else if (codepoint <= 0x7ff) {
2471 out.append(1, static_cast<target_t>(0xc0 | ((codepoint >> 6) & 0x1f)));
2472 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
2473 } else if (codepoint <= 0xffff) {
2474 out.append(1, static_cast<target_t>(0xe0 | ((codepoint >> 12) & 0x0f)));
2475 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 6) & 0x3f)));
2476 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
2477 } else {
2478 out.append(1, static_cast<target_t>(0xf0 | ((codepoint >> 18) & 0x07)));
2479 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 12) & 0x3f)));
2480 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 6) & 0x3f)));
2481 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
2482 }

2484 }

2682 auto codepoint = 0u;
2683 for (const auto& character : str) {
2684 if (character >= 0xd800 && character <= 0xdbff)
2685 codepoint = ((character - 0xd800) << 10) + 0x10000;
2686 else {
2687 if (character >= 0xdc00 && character <= 0xdfff) codepoint |= character - 0xdc00;
2688 else codepoint = character;
2689
2690 if (codepoint <= 0x7f)
2691 out.append(1, static_cast<target_t>(codepoint));
2692 else if (codepoint <= 0x7ff) {

2694 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
2695 } else if (codepoint <= 0xffff) {
2696 out.append(1, static_cast<target_t>(0xe0 | ((codepoint >> 12) & 0x0f)));

2698 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
