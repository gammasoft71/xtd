1497inline std::basic_string<target_t> __xtd_convert_to_string(const std::basic_string<source_t>& str) noexcept {
1498 auto out = std::basic_string<target_t> {};
1499 auto codepoint = 0u;
1500 for (const auto& character : str) {
1501 if (character >= 0xd800 && character <= 0xdbff)
1502 codepoint = ((character - 0xd800) << 10) + 0x10000;
1503 else {
1504 if (character >= 0xdc00 && character <= 0xdfff) codepoint |= character - 0xdc00;
1505 else codepoint = character;
1506
1507 if (codepoint <= 0x7f)
1508 out.append(1, static_cast<target_t>(codepoint));
1509 else if (codepoint <= 0x7ff) {
1510 out.append(1, static_cast<target_t>(0xc0 | ((codepoint >> 6) & 0x1f)));
1511 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
1512 } else if (codepoint <= 0xffff) {
1513 out.append(1, static_cast<target_t>(0xe0 | ((codepoint >> 12) & 0x0f)));
1514 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 6) & 0x3f)));
1515 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
1516 } else {
1517 out.append(1, static_cast<target_t>(0xf0 | ((codepoint >> 18) & 0x07)));
1518 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 12) & 0x3f)));
1519 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 6) & 0x3f)));
1520 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));

1523 }
