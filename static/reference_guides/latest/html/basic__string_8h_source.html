1475inline std::basic_string<target_t> __xtd_convert_to_string(const std::basic_string<source_t>& str) noexcept {
1476 auto out = std::basic_string<target_t> {};
1477 auto codepoint = 0u;
1478 for (const auto& character : str) {
1479 if (character >= 0xd800 && character <= 0xdbff)
1480 codepoint = ((character - 0xd800) << 10) + 0x10000;
1481 else {
1482 if (character >= 0xdc00 && character <= 0xdfff) codepoint |= character - 0xdc00;
1483 else codepoint = character;
1484
1485 if (codepoint <= 0x7f)
1486 out.append(1, static_cast<target_t>(codepoint));
1487 else if (codepoint <= 0x7ff) {
1488 out.append(1, static_cast<target_t>(0xc0 | ((codepoint >> 6) & 0x1f)));
1489 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
1490 } else if (codepoint <= 0xffff) {
1491 out.append(1, static_cast<target_t>(0xe0 | ((codepoint >> 12) & 0x0f)));
1492 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 6) & 0x3f)));
1493 out.append(1, static_cast<target_t>(0x80 | (codepoint & 0x3f)));
1494 } else {
1495 out.append(1, static_cast<target_t>(0xf0 | ((codepoint >> 18) & 0x07)));
1496 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 12) & 0x3f)));
1497 out.append(1, static_cast<target_t>(0x80 | ((codepoint >> 6) & 0x3f)));
