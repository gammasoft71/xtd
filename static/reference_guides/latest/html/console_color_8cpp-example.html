console_color.cpp

Shows how to use xtd::console::background_color and xtd::console::foreground_color methods.

1#include <xtd/console>
2
3using namespace xtd;
4
5auto main() -> int {
6 // Get an array with the values of console_color enumeration members.
7 auto colors = {console_color::black, console_color::dark_blue, console_color::dark_green, console_color::dark_cyan, console_color::dark_red, console_color::dark_magenta, console_color::dark_yellow, console_color::gray, console_color::dark_gray, console_color::blue, console_color::green, console_color::cyan, console_color::red, console_color::magenta, console_color::yellow, console_color::white};
8
9 // Save the current background and foreground colors.
10 auto current_background = console::background_color();
11 auto current_foreground = console::foreground_color();
12
13 // Display all foreground colors except the one that matches the background.
14 console::write_line("All the foreground colors except {}, the background color:", current_background);
15 for (auto color : colors) {
16 if (color == current_background) continue;
17
18 console::foreground_color(color);
19 console::write_line(" The foreground color is {}.", color);
20 }
21 console::write_line();
22 // Restore the foreground color.
23 console::foreground_color(current_foreground);
24
25 // Display each background color except the one that matches the current foreground color.
26 console::write_line("All the background colors except {}, the foreground color:", current_foreground);
27 for (auto color : colors) {
28 if (color == current_foreground) continue;
29
30 console::background_color(color);
31 console::write_line(" The background color is {}.", color);
32 }
33
34 // Restore the original console colors.
35 console::reset_color();
36 console::write_line("\nOriginal colors restored...");
37}
