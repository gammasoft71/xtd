7namespace environment_cancel_sgnal_example {
8 class program static_ {
9 public:
10 // The main entry point for the application.

17 show_help();

19 console::treat_control_c_as_input(true);
20 while (true) {
21 if (console::key_available()) {
22 auto key_info = console::read_key(true);
23 if (key_info.modifiers() == console_modifiers::none && key_info.key() == console_key::escape) exit();
24 else if (key_info.modifiers() == console_modifiers::control && key_info.key() == console_key::c) generate_signal(signal::interrupt);
25 else switch (key_info.key_char()) {
26 case 'a': generate_signal(signal::abnormal_termination); break;
27 case 'c': generate_signal(signal::interrupt); break;
28 case 'e': exit(); break;
29 case 'f': generate_signal(signal::floating_point_exception); break;
30 case 'h': show_help(); break;
31 case 'i': generate_signal(signal::illegal_instruction); break;
32 case 't': generate_signal(signal::software_termination); break;
33 case 'u': cancel = !cancel; show_help(); break;
34 case 'v': generate_signal(signal::segmentation_violation); break;
35 default: break;
36 }
37 }
38 }
39 }

41 private:
