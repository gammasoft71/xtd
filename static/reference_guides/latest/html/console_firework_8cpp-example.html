132 class program static_ {
133 public:
134 static void main() {
135 console::write_line("Adjuste window size and pres any key when ready to start...");
136 auto console_height = console::window_height();
137 auto console_width = console::window_width();
138 console::read_key(false);
139 console::cursor_visible(false);
140 console::clear();
141
142 auto rand = xtd::random {};
143 auto fireworks = std::list<ptr<firework>> {};
144 auto colors = list {console_color::blue, console_color::green, console_color::cyan, console_color::red, console_color::magenta, console_color::yellow, console_color::white};
145
146 while (!console::key_available()) {
147 fireworks.push_back(new_ptr<firework_start>(rand.next(2, console_width - 2), rand.next(2, console_height - 2), colors[rand.next(colors.size())], rand.next(1, 5)));
148
149 auto fireworks_to_removed = list<ptr<firework>> {};
150 for (auto& firework : fireworks) {
151 if (is<firework_end>(firework)) fireworks_to_removed.push_back(firework);
152 explode(firework);
153 }
154
155 for (auto& firework : fireworks_to_removed)
156 fireworks.remove(firework);
157
158 threading::thread::sleep(100_ms);
159 }
160
161 console::reset_color();
162 console::clear();
163 console::cursor_visible(true);
164 }
165
166 private:
167 static void explode(ptr<firework>& firework) {
168 firework->paint();
169 if (is<firework_start_expanded4>(firework)) firework = new_ptr<firework_end>(*firework);
170 if (is<firework_start_expanded3>(firework)) firework = new_ptr<firework_start_expanded4>(*firework);
171 if (is<firework_start_expanded2>(firework)) firework = new_ptr<firework_start_expanded3>(*firework);
172 if (is<firework_start_expanded1>(firework)) firework = new_ptr<firework_start_expanded2>(*firework);
173 if (is<firework_exploded>(firework)) firework = new_ptr<firework_start_expanded1>(*firework);
174 if (date_time::now().ticks() % firework->delay() == 0 && is<firework_start>(firework)) firework = new_ptr<firework_exploded>(*firework);
175 }
176 };
177}
