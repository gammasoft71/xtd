133 void form_main_key_press(object& sender, key_press_event_args& e) {
134 if (e.key_char() >= '0' && e.key_char() <= '9') {
135 button_numbers[e.key_char() - '0'].perform_click();
136 e.handled(true);
137 return;
138 }
139 switch (e.key_char()) {
140 case 'c' : button_clear.perform_click(); e.handled(true); break;
141 case 'C' : button_clear.perform_click(); e.handled(true); break;
142 case '%' : button_percent.perform_click(); e.handled(true); break;
143 case ',' : button_decimal.perform_click(); e.handled(true); break;
144 case '.' : button_decimal.perform_click(); e.handled(true); break;
145 case '/' : button_operators[as<int>(operators::divide)].perform_click(); e.handled(true); break;
146 case '*' : button_operators[as<int>(operators::multiply)].perform_click(); e.handled(true); break;
147 case '-' : button_operators[as<int>(operators::subtract)].perform_click(); e.handled(true); break;
148 case '+' : button_operators[as<int>(operators::add)].perform_click(); e.handled(true); break;
149 case '=' : button_operators[as<int>(operators::equal)].perform_click(); e.handled(true); break;
150 case 13 : button_operators[as<int>(operators::equal)].perform_click(); e.handled(true); break;
151 case 127 : back_space_key_press(sender, e); e.handled(true); break;
152 }
153 }

183 if (!first_operand.has_value()) first_operand = parse<double>(result.text());

186 switch (operation) {

196 operation = map<string, operators> {{"÷", operators::divide}, {"x", operators::multiply}, {"-", operators::subtract}, {"+", operators::add}, {"=", operation}} [as<control>(sender).text()];
