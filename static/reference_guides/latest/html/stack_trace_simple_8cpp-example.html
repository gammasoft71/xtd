stack_trace_simple.cpp

Demonstrates how to create a simple xtd::diagnostics::stack_trace class and iterate through its frames to obtain debugging and diagnostic information.

1#include <xtd/diagnostics/stack_trace>
2#include <xtd/console>
3
4using namespace xtd;
5using namespace xtd::diagnostics;
6
7class stack_trace_sample {
8public:
9 void my_public_method() {
10 my_protected_method();
11 }
12
13protected:
14 void my_protected_method() {
15 auto mic = my_internal_class {};
16 mic.throws_exception();
17 }
18
19private:
20 class my_internal_class {
21 public:
22 void throws_exception() {
23 try {
24 throw system_exception("A problem was encountered.");
25 } catch (const system_exception&) {
26 // Create a StackTrace that captures filename, line number and column information.
27 auto st = stack_trace {true};
28 ustring stack_indent = "";
29 for (auto i = 0ul; i < st.frame_count(); i++) {
30 // Note that at this level, there are four stack frames, one for each method invocation.
31 auto sf = st.get_frame(i);
32 console::write_line();
33 console::write_line(stack_indent + " Method: {0}", sf.get_method());
34 console::write_line(stack_indent + " File: {0}", sf.get_file_name());
35 console::write_line(stack_indent + " Line Number: {0}", sf.get_file_line_number());
36 stack_indent += " ";
37 }
38 throw;
39 }
40 }
41 };
42};
43
44auto main() -> int {
45 auto sample = stack_trace_sample {};
46 try {
47 sample.my_public_method();
48 } catch (const system_exception&) {
49 // Create a StackTrace that captures filename, line number, and column information for the current thread.
50 auto st = stack_trace {true};
51 for (auto i = 0ul; i < st.frame_count(); i++) {
52 // Note that high up the call stack, there is only one stack frame.
53 auto sf = st.get_frame(i);
54 console::write_line();
55 console::write_line("High up the call stack, Method: {0}", sf.get_method());
56
57 console::write_line("High up the call stack, Line Number: {0}", sf.get_file_line_number());
58 }
59 }
60}
