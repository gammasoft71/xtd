26class signal_generator_base abstract_ {
27public:
28 ~signal_generator_base() {enabled(false);}
29
30 bool enabled() const noexcept {return enabled_;}
31 signal_generator_base& enabled(bool value) noexcept {
32 if (enabled_ == value) return *this;
33 enabled_ = value;
34 if (enabled_) generator_result_ = generator_.begin_invoke();
35 else generator_.end_invoke(generator_result_);
36 return *this;
37 }
38
39 const time_span& interval() const noexcept {return interval_;}
40 signal_generator_base& interval(const time_span& value) noexcept {
41 interval_ = value;
42 return *this;
43 }
44
45 event<signal_generator_base, signal_generated_event_handler> signal_generated;
46
47 void start() noexcept {enabled(true);}
48 void stop() noexcept {enabled(false);}
49
50protected:
51 void on_signal_generated(const signal_generated_event_args& e) {signal_generated(*this, e);}
52
53 virtual float next_signal() noexcept = 0;
54
55private:
56 bool enabled_ = false;
57 time_span interval_ = 100_ms;
58 delegate<void()> generator_ {[&] {
59 while(enabled_) {
60 threading::thread::sleep(interval_);
61 on_signal_generated({next_signal(), date_time::now()});
62 }
63 }};
