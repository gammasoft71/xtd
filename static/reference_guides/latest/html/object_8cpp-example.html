3using namespace xtd;

5// The Point class is derived from System.Object.
6class point : public object {
7public:
8 int x, y;
9
10 point(int x, int y) : x {x}, y {y} {}
11
12 bool equals(const object& obj) const noexcept override {
13 // If this and obj do not refer to the same type, then they are not equal.
14 if (obj.get_type() != get_type()) return false;
15
16 // Return true if x and y fields match.
17 auto& other = static_cast<const point&>(obj);
18 return x == other.x && y == other.y;
19 }
20
21 // Return the XOR of the x and y fields.
22 size_t get_hash_code() const noexcept override {
23 return x ^ y;
24 }
25
26 // Return the point's value as a string.
27 ustring to_string() const noexcept override {
28 return ustring::format("({}, {})", x, y);
29 }
30
31 // Return a copy of this point object by making a simple field copy.
32 uptr<point> copy() const {
33 return memberwise_clone<point>();
34 }
35};
36
37auto main() -> int {
38 // Construct a Point object.
39 auto p1 = point {1, 2};
40
41 // Make another Point object that is a copy of the first.
42 auto p2 = p1.copy();
43
44 // Make another variable that references the first point object.
45 auto& p3 = p1;
46
47 // The line below displays false because p1 and p2 refer to two different objects.
48 console::write_line(object::reference_equals(p1, *p2));
49
50 // The line below displays true because p1 and p2 refer to two different objects that have the same value.
51 console::write_line(object::equals(p1, *p2));
52
53 // The line below displays true because p1 and p3 refer to one object.
54 console::write_line(object::reference_equals(p1, p3));
55
56 // The line below displays: p1's value is: (1, 2)
57 console::write_line("p1's value is: {}", p1.to_string());
58}
