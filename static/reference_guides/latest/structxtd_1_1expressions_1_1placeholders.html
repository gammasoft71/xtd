#include <xtd/xtd>
auto main() -> int {
//auto expr1 = [](auto&& operand_one, auto&& operand_two) {return operand_one + operand_two;};

//auto expr2 = [](auto&& _1, auto&& _2) {return _1 + _2;};

}
// This code produces the following output :

// expr1 result => 30
// expr2 result => 30
