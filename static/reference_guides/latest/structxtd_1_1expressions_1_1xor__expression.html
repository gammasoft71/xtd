Examples
The following example shows how to use xtd::expressions::xor_expression.
#include <xtd/xtd>
auto main() -> int {
//auto bit_xor1 = [](auto&& _) {return _ ^ 0x0F;};

println("bit_xor1 result => {}", bit_xor1(42));
//auto bit_xor2 = [](auto&& _1, auto&& _2) {return _1 ^ _2;};

}
