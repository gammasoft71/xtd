#include <xtd/console>
#include <xtd/environment>
using namespace xtd;
void test3() {
console::write_line("environment::stack_trace:");
console::write_line(environment::stack_trace());
}
void test2() {
test3();
}
void test1() {
test2();
}
auto main() -> int {
test1();
}
