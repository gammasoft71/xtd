directory_assert_are_not_equal.cpp

Shows how to use xtd::tunit::directory_assert::are_not_equal method.

#include <xtd/xtd.tunit>
using namespace xtd;
using namespace xtd::io;
using namespace xtd::tunit;
namespace unit_tests {
class test_class_(test) {
void test_method_(test_case_succeed) {
auto di = directory_info {environment::get_folder_path(environment::special_folder::system)};
directory_assert::are_not_equal(directory_info {path::get_temp_path()}, di);
}
void test_method_(test_case_failed) {
auto di = directory_info {path::get_temp_path()};
directory_assert::are_not_equal(directory_info {path::get_temp_path()}, di);
}
};
}
auto main() -> int {
return console_unit_test().run();
}
